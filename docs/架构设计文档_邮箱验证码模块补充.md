## 邮箱验证码模块设计

### 1. 功能概述

邮箱验证码模块用于以下场景：
- **用户注册**：验证邮箱有效性，确保用户拥有该邮箱
- **忘记密码**：验证用户身份，允许重置密码

### 2. 技术选型

| 组件 | 选型 | 说明 |
|------|------|------|
| 邮箱服务 | 163邮箱SMTP | 稳定可靠，国内用户友好 |
| 发送协议 | SMTP | 标准邮件发送协议 |
| 验证码存储 | Redis | 设置过期时间，自动清理 |
| 验证码格式 | 6位数字 | 安全性与易用性平衡 |

### 3. 163邮箱配置

#### 3.1 开启SMTP服务

1. 登录163邮箱 -> 设置 -> POP3/SMTP/IMAP
2. 开启"POP3/SMTP服务"
3. 获取**授权码**（不是邮箱密码）

#### 3.2 SMTP配置参数

```
SMTP服务器: smtp.163.com
端口: 465 (SSL加密) 或 25 (非加密，不推荐)
用户名: your-email@163.com
密码: 授权码（非邮箱登录密码）
```

### 7. 后端实现

#### 7.1 邮件发送服务 (email_service.py)

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.utils import formataddr
import redis
import random
import string
from datetime import datetime, timedelta

class EmailService:
    def __init__(self, smtp_host: str, smtp_port: int, 
                 smtp_user: str, smtp_password: str,
                 from_name: str, redis_client: redis.Redis):
        self.smtp_host = smtp_host
        self.smtp_port = smtp_port
        self.smtp_user = smtp_user
        self.smtp_password = smtp_password
        self.from_name = from_name
        self.redis = redis_client
    
    def generate_code(self, length: int = 6) -> str:
        """生成随机验证码"""
        return ''.join(random.choices(string.digits, k=length))
    
    def send_verify_code(self, to_email: str, code: str, code_type: str) -> bool:
        """发送验证码邮件"""
        try:
            # 构建邮件内容
            msg = MIMEMultipart('alternative')
            msg['From'] = formataddr((self.from_name, self.smtp_user))
            msg['To'] = to_email
            
            if code_type == 'register':
                msg['Subject'] = '【Aigc For Study】注册验证码'
                template = self._get_register_template(code)
            else:
                msg['Subject'] = '【Aigc For Study】密码重置验证码'
                template = self._get_reset_password_template(code)
            
            msg.attach(MIMEText(template, 'html', 'utf-8'))
            
            # 发送邮件
            with smtplib.SMTP_SSL(self.smtp_host, self.smtp_port) as server:
                server.login(self.smtp_user, self.smtp_password)
                server.sendmail(self.smtp_user, [to_email], msg.as_string())
            
            return True
        except Exception as e:
            print(f"发送邮件失败: {e}")
            return False
    
    def _get_register_template(self, code: str) -> str:
        """注册验证码邮件模板"""
        return f"""
        <div style="max-width: 600px; margin: 0 auto; padding: 20px; font-family: Arial, sans-serif;">
            <h2 style="color: #333; text-align: center;">欢迎注册 Aigc For Study</h2>
            <p style="color: #666;">您好！</p>
            <p style="color: #666;">您正在注册 Aigc For Study 账号，验证码如下：</p>
            <div style="background-color: #f5f5f5; padding: 20px; text-align: center; margin: 20px 0;">
                <span style="font-size: 32px; font-weight: bold; color: #1890ff; letter-spacing: 5px;">{code}</span>
            </div>
            <p style="color: #999; font-size: 12px;">验证码有效期为5分钟，请尽快完成验证。</p>
            <p style="color: #999; font-size: 12px;">如果您没有进行此操作，请忽略此邮件。</p>
            <hr style="border: none; border-top: 1px solid #eee; margin: 30px 0;">
            <p style="color: #999; font-size: 12px; text-align: center;">
                此邮件由系统自动发送，请勿直接回复。
            </p>
        </div>
        """
    
    def _get_reset_password_template(self, code: str) -> str:
        """重置密码验证码邮件模板"""
        return f"""
        <div style="max-width: 600px; margin: 0 auto; padding: 20px; font-family: Arial, sans-serif;">
            <h2 style="color: #333; text-align: center;">密码重置验证</h2>
            <p style="color: #666;">您好！</p>
            <p style="color: #666;">您正在重置 Aigc For Study 账号密码，验证码如下：</p>
            <div style="background-color: #f5f5f5; padding: 20px; text-align: center; margin: 20px 0;">
                <span style="font-size: 32px; font-weight: bold; color: #ff4d4f; letter-spacing: 5px;">{code}</span>
            </div>
            <p style="color: #999; font-size: 12px;">验证码有效期为5分钟，请尽快完成验证。</p>
            <p style="color: #ff4d4f; font-size: 12px;">如果您没有进行此操作，请立即修改密码以保护账号安全。</p>
            <hr style="border: none; border-top: 1px solid #eee; margin: 30px 0;">
            <p style="color: #999; font-size: 12px; text-align: center;">
                此邮件由系统自动发送，请勿直接回复。
            </p>
        </div>
        """
    
    def store_code(self, email: str, code: str, code_type: str, 
                   expire_seconds: int = 300) -> bool:
        """存储验证码到Redis"""
        key = f"verify_code:{code_type}:{email}"
        return self.redis.setex(key, expire_seconds, code)
    
    def verify_code(self, email: str, code: str, code_type: str) -> bool:
        """验证验证码"""
        key = f"verify_code:{code_type}:{email}"
        stored_code = self.redis.get(key)
        
        if stored_code and stored_code.decode() == code:
            # 验证成功后删除验证码
            self.redis.delete(key)
            return True
        return False
    
    def check_cooldown(self, email: str, code_type: str, 
                       cooldown_seconds: int = 60) -> bool:
        """检查发送冷却时间"""
        key = f"verify_code_cooldown:{code_type}:{email}"
        return self.redis.exists(key)
    
    def set_cooldown(self, email: str, code_type: str, 
                     cooldown_seconds: int = 60):
        """设置发送冷却时间"""
        key = f"verify_code_cooldown:{code_type}:{email}"
        self.redis.setex(key, cooldown_seconds, "1")
```

```python
from fastapi import APIRouter, HTTPException, Request
from pydantic import BaseModel, EmailStr
from app.services.email_service import EmailService
from app.core.config import settings
from app.core.database import get_db
from app.models.user import User
from app.utils.security import hash_password, verify_password

router = APIRouter(prefix="/auth", tags=["认证"])

class SendVerifyCodeRequest(BaseModel):
    email: EmailStr
    type: str  # register 或 reset_password

class RegisterRequest(BaseModel):
    email: EmailStr
    password: str
    verify_code: str
    role: str

class ForgotPasswordRequest(BaseModel):
    email: EmailStr
    verify_code: str
    new_password: str

@router.post("/send-verify-code")
async def send_verify_code(request: SendVerifyCodeRequest, req: Request):
    """发送验证码"""
    email_service = EmailService(
        smtp_host=settings.SMTP_HOST,
        smtp_port=settings.SMTP_PORT,
        smtp_user=settings.SMTP_USER,
        smtp_password=settings.SMTP_PASSWORD,
        from_name=settings.SMTP_FROM_NAME,
        redis_client=settings.redis
    )
    
    # 检查发送冷却时间
    if email_service.check_cooldown(request.email, request.type):
        raise HTTPException(
            status_code=429, 
            detail="发送过于频繁，请稍后再试"
        )
    
    # 检查邮箱是否已注册（注册场景）
    if request.type == "register":
        db = get_db()
        existing_user = db.query(User).filter(
            User.email == request.email
        ).first()
        if existing_user:
            raise HTTPException(
                status_code=400, 
                detail="该邮箱已被注册"
            )
    
    # 检查邮箱是否存在（重置密码场景）
    if request.type == "reset_password":
        db = get_db()
        existing_user = db.query(User).filter(
            User.email == request.email
        ).first()
        if not existing_user:
            raise HTTPException(
                status_code=400, 
                detail="该邮箱未注册"
            )
    
    # 生成验证码
    code = email_service.generate_code(settings.VERIFY_CODE_LENGTH)
    
    # 存储验证码
    email_service.store_code(
        request.email, 
        code, 
        request.type,
        settings.VERIFY_CODE_EXPIRE_SECONDS
    )
    
    # 发送邮件
    success = email_service.send_verify_code(
        request.email, 
        code, 
        request.type
    )
    
    if not success:
        raise HTTPException(
            status_code=500, 
            detail="验证码发送失败，请稍后重试"
        )
    
    # 设置冷却时间
    email_service.set_cooldown(
        request.email, 
        request.type,
        settings.VERIFY_CODE_COOLDOWN_SECONDS
    )
    
    return {
        "code": 200,
        "message": "验证码已发送，请查收邮件",
        "data": {
            "expire_seconds": settings.VERIFY_CODE_EXPIRE_SECONDS,
            "cooldown_seconds": settings.VERIFY_CODE_COOLDOWN_SECONDS
        }
    }
```

### 8. 安全策略

#### 8.1 防刷机制

| 策略 | 说明 | 目的 |
|------|------|------|
| 发送频率限制 | 同一邮箱60秒内只能发送一次 | 防止短时间重复发送 |
| IP频率限制 | 同一IP每分钟最多发送3次 | 防止IP滥用 |
| 验证码有效期 | 5分钟，过期自动失效 | 限制验证码有效时间 |
| 设备验证失败锁定 | 同一设备验证失败5次后锁定15分钟 | 防止暴力破解 |
| 图形验证码 | 发送前需通过图形验证码 | 防止自动化攻击 |

**注意：**
- 不设置每日上限，避免攻击者故意触发上限阻止正常用户使用
- 验证失败锁定基于设备特征，而非邮箱，避免影响正常用户

#### 8.2 设备指纹生成

由于Web应用无法直接获取设备MAC地址，采用多特征组合生成设备指纹：

```javascript
// 前端设备指纹生成 (deviceFingerprint.js)
export function generateDeviceFingerprint() {
  const components = [];
  
  // 1. User-Agent
  components.push(navigator.userAgent);
  
  // 2. 屏幕信息
  components.push(`${screen.width}x${screen.height}x${screen.colorDepth}`);
  
  // 3. 时区
  components.push(Intl.DateTimeFormat().resolvedOptions().timeZone);
  
  // 4. 语言
  components.push(navigator.language);
  
  // 5. 平台
  components.push(navigator.platform);
  
  // 6. Canvas指纹
  components.push(getCanvasFingerprint());
  
  // 7. WebGL指纹
  components.push(getWebGLFingerprint());
  
  // 8. 字体列表（部分浏览器支持）
  components.push(getFontFingerprint());
  
  // 生成SHA256哈希
  const fingerprint = sha256(components.join('|'));
  
  return fingerprint;
}

function getCanvasFingerprint() {
  try {
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    ctx.textBaseline = 'top';
    ctx.font = '14px Arial';
    ctx.fillStyle = '#f60';
    ctx.fillRect(125, 1, 62, 20);
    ctx.fillStyle = '#069';
    ctx.fillText('Device Fingerprint', 2, 15);
    return canvas.toDataURL();
  } catch (e) {
    return 'canvas-unsupported';
  }
}

function getWebGLFingerprint() {
  try {
    const canvas = document.createElement('canvas');
    const gl = canvas.getContext('webgl');
    const debugInfo = gl.getExtension('WEBGL_debug_renderer_info');
    return `${gl.getParameter(debugInfo.UNMASKED_VENDOR_WEBGL)}|${gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL)}`;
  } catch (e) {
    return 'webgl-unsupported';
  }
}

function getFontFingerprint() {
  // 简化版字体检测
  const baseFonts = ['monospace', 'sans-serif', 'serif'];
  const testFonts = ['Arial', 'Verdana', 'Times New Roman', 'Courier New'];
  // 实际实现需要更复杂的字体检测逻辑
  return testFonts.filter(font => isFontAvailable(font)).join(',');
}

function sha256(str) {
  // 使用Web Crypto API
  return crypto.subtle.digest('SHA-256', new TextEncoder().encode(str))
    .then(hash => Array.from(new Uint8Array(hash))
      .map(b => b.toString(16).padStart(2, '0')).join(''));
}
```

#### 8.3 Redis存储结构

```
# 验证码存储
verify_code:register:user@example.com -> "123456" (TTL: 300s)

# 冷却时间（防止短时间重复发送）
verify_code_cooldown:register:user@example.com -> "1" (TTL: 60s)

# IP发送计数（每分钟重置）
verify_code_ip:192.168.1.1 -> "2" (TTL: 60s)

# 设备验证失败计数（基于设备指纹）
verify_code_fail:device:a1b2c3d4e5f6... -> "3" (TTL: 900s)

# 设备验证锁定（失败5次后锁定）
verify_code_lock:device:a1b2c3d4e5f6... -> "1" (TTL: 900s)
```

#### 8.4 防刷逻辑实现

```python
import hashlib

class DeviceFingerprintService:
    def __init__(self, redis_client):
        self.redis = redis_client
    
    def generate_device_id(self, request) -> str:
        """
        服务端生成设备ID（基于请求特征）
        """
        components = [
            request.client.host,  # IP地址
            request.headers.get('user-agent', ''),
            request.headers.get('accept-language', ''),
            request.headers.get('sec-ch-ua', ''),  # Chrome客户端提示
            request.headers.get('sec-ch-ua-platform', ''),
        ]
        
        # 如果前端传来了设备指纹，也加入计算
        client_fingerprint = request.headers.get('x-device-fingerprint', '')
        if client_fingerprint:
            components.append(client_fingerprint)
        
        # 生成设备ID
        device_string = '|'.join(components)
        return hashlib.sha256(device_string.encode()).hexdigest()[:32]
    
    def check_rate_limit(self, email: str, device_id: str, code_type: str) -> tuple[bool, str]:
        """
        检查发送频率限制
        返回: (是否允许发送, 错误信息)
        """
        # 1. 检查邮箱冷却时间
        cooldown_key = f"verify_code_cooldown:{code_type}:{email}"
        if self.redis.exists(cooldown_key):
            ttl = self.redis.ttl(cooldown_key)
            return False, f"请{ttl}秒后再试"
        
        # 2. 检查IP频率（每分钟最多3次）- 通过device_id间接实现
        ip_key = f"verify_code_ip:{device_id[:16]}"  # 使用device_id前16位作为IP相关标识
        ip_count = self.redis.get(ip_key)
        if ip_count and int(ip_count) >= 3:
            return False, "发送过于频繁，请稍后再试"
        
        # 3. 检查设备验证失败锁定（防止暴力破解）
        lock_key = f"verify_code_lock:device:{device_id}"
        if self.redis.exists(lock_key):
            ttl = self.redis.ttl(lock_key)
            return False, f"该设备验证失败次数过多，请{ttl}秒后再试"
        
        return True, ""
    
    def record_send(self, email: str, device_id: str, code_type: str):
        """记录发送行为"""
        # 设置冷却时间
        cooldown_key = f"verify_code_cooldown:{code_type}:{email}"
        self.redis.setex(cooldown_key, 60, "1")
        
        # 增加设备计数
        ip_key = f"verify_code_ip:{device_id[:16]}"
        if self.redis.exists(ip_key):
            self.redis.incr(ip_key)
        else:
            self.redis.setex(ip_key, 60, "1")
    
    def record_verify_fail(self, device_id: str, code_type: str):
        """记录验证失败（基于设备）"""
        fail_key = f"verify_code_fail:device:{device_id}"
        
        if self.redis.exists(fail_key):
            count = self.redis.incr(fail_key)
        else:
            count = 1
            self.redis.setex(fail_key, 900, "1")  # 15分钟
        
        # 失败5次后锁定该设备
        if count >= 5:
            lock_key = f"verify_code_lock:device:{device_id}"
            self.redis.setex(lock_key, 900, "1")  # 锁定15分钟
            self.redis.delete(fail_key)  # 清除失败计数
    
    def clear_verify_fail(self, device_id: str, code_type: str):
        """验证成功后清除失败计数"""
        fail_key = f"verify_code_fail:device:{device_id}"
        self.redis.delete(fail_key)
```
